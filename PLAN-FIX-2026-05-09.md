### 2. Cross-platform discipline (Windows + Mac)

**Cod nou MUST respecta:**

```js
// CORECT
import { join } from 'node:path';
import { homedir } from 'node:os';
const p = join(homedir(), '.claude', 'projects');

// GRESIT (shellism)
const p = `${process.env.HOME}/.claude/projects`;  // Windows: HOME e undefined
const p = exec('ls path/');                         // Windows: nu merge
```

**Reguli explicite:**

| Surafata | Windows | Mac | Decizie |
|----------|---------|-----|---------|
| Path separator | `\` | `/` | INTOTDEAUNA `path.join` |
| Home dir | `%USERPROFILE%` | `$HOME` | INTOTDEAUNA `os.homedir()` |
| Shell scripts | `.cmd` / `.ps1` | `.sh` | Sursa = `.js` cross-platform; wrappers thin |
| Line endings | CRLF default | LF default | `.gitattributes` cu `* text=auto eol=lf` + override `*.cmd,*.ps1 eol=crlf` |
| File encoding | UTF-8 cu BOM uneori | UTF-8 | `writeFileSync(..., { encoding: 'utf8' })` explicit |
| Path comparison | Case-insensitive | Case-sensitive | Niciodata compara path-uri prin `===` direct; foloseste `path.relative` + check empty |
| File locks | NTFS strict (EBUSY pe rename-over-open) | rar | Toate atomic writes: try/catch + cleanup tmp + retry-with-backoff |
| Process spawning | `cmd.exe` / `pwsh` | `bash` / `zsh` | INTOTDEAUNA `spawn(node, ['scripts/X.js'])` cu `shell: false` |
| Network bind | `0.0.0.0` necesita admin uneori | open | Default `127.0.0.1`, document risk daca schimbi |
| Symlink behavior | Limitat (necesita admin/dev mode) | OK | Evit symlinks in distributie |

**Verificare automata cross-platform** (Faza 0 task):
- Lint custom in `scripts/lint-portability.js` care esueaza pe: hardcoded `/`, hardcoded `\`, `process.env.HOME`, `os.tmpdir()` cu suffixe shellish, etc.
- Smoke runner ruleaza pe ambele: pe Windows host pe loc; pe Mac via simulare prin `process.platform = 'darwin'` mocking acolo unde e relevant.

```js
// scripts/lib/env-loader.js
import { readFileSync, existsSync } from 'node:fs';
import { fileURLToPath } from 'node:url';
import { dirname, join, resolve } from 'node:path';

const __filename = fileURLToPath(import.meta.url);
const ROBOS_ROOT = resolve(dirname(__filename), '..', '..');

let _loaded = false;

/**
 * Idempotent .env loader for hook scripts.
 * Loads ROBOS_ROOT/.env into process.env if not already loaded.
 * Does NOT throw on missing file (graceful degradation).
 * Already-set process.env wins (no overwrite).
 */
export function loadEnv() {
  if (_loaded) return;
  _loaded = true;

  const envPath = join(ROBOS_ROOT, '.env');
  if (!existsSync(envPath)) return;

  let content;
  try {
    content = readFileSync(envPath, 'utf8');
  } catch {
    return; // .env locked / unreadable — graceful
  }

  for (const line of content.split(/\r?\n/)) {
    const trimmed = line.trim();
    if (!trimmed || trimmed.startsWith('#')) continue;
    const eq = trimmed.indexOf('=');
    if (eq < 0) continue;

    const key = trimmed.slice(0, eq).trim();
    let value = trimmed.slice(eq + 1).trim();

    // Strip optional surrounding quotes
    if ((value.startsWith('"') && value.endsWith('"')) ||
        (value.startsWith("'") && value.endsWith("'"))) {
      value = value.slice(1, -1);
    }

    // process.env wins (don't overwrite parent-set values)
    if (process.env[key] === undefined) {
      process.env[key] = value;
    }
  }
}
```

**De ce nou shared lib si nu reuse `centre/lib/env-loader.js`:** centre version pulleaza dependinte server-side (caching, validation). Hook scripts trebuie sa fie ultra-light (run on every prompt). Niciodata import `centre/*` din `scripts/*`.

Creeaza `scripts/lib/atomic-write.js`:

```js
import { writeFileSync, renameSync, unlinkSync, existsSync } from 'node:fs';
import { randomBytes } from 'node:crypto';

/**
 * Atomic write with cleanup on failure.
 * Cross-platform: handles Windows EBUSY/EPERM with retry.
 */
export function atomicWrite(targetPath, content, options = {}) {
  const { encoding = 'utf8', mode = 0o644, retries = 3, retryDelayMs = 50 } = options;
  const tmpPath = `${targetPath}.${randomBytes(4).toString('hex')}.tmp`;

  try {
    writeFileSync(tmpPath, content, { encoding, mode });

    let lastErr;
    for (let i = 0; i < retries; i++) {
      try {
        renameSync(tmpPath, targetPath);
        return;
      } catch (err) {
        lastErr = err;
        if (err.code !== 'EBUSY' && err.code !== 'EPERM') throw err;
        // Windows: file may be temporarily locked, sleep and retry
        const start = Date.now();
        while (Date.now() - start < retryDelayMs) { /* spin */ }
      }
    }
    throw lastErr;
  } catch (err) {
    // Cleanup tmp on any error
    if (existsSync(tmpPath)) {
      try { unlinkSync(tmpPath); } catch { /* best effort */ }
    }
    throw err;
  }
}
```

Apoi inlocuieste atomicWrite locale din loop-detector + client-context cu import din lib comun.

**Cross-platform:** retry-logic doar pe EBUSY/EPERM (Windows-only codes; pe Mac nu se intampla → instant pass).

**Implementare:**

```js
// scripts/lib/cleanup.js
import { readdirSync, statSync, unlinkSync } from 'node:fs';
import { join } from 'node:path';

export function pruneDirByAge(dir, maxAgeDays, predicate = () => true) {
  if (!existsSync(dir)) return { removed: 0, kept: 0 };
  const cutoffMs = Date.now() - maxAgeDays * 86400_000;
  let removed = 0, kept = 0;
  for (const name of readdirSync(dir)) {
    const p = join(dir, name);
    try {
      const st = statSync(p);
      if (st.mtimeMs < cutoffMs && predicate(name, st)) {
        unlinkSync(p);
        removed++;
      } else {
        kept++;
      }
    } catch { /* skip */ }
  }
  return { removed, kept };
}
```

In `session-timeout-detector.js`, dupa scan-ul existent:

```js
import { pruneDirByAge } from './lib/cleanup.js';

const sessionStateDir = join(ROBOS_ROOT, 'data', 'session-state');
const recoveryDir = join(ROBOS_ROOT, 'data', 'session-recovery');

const sst = pruneDirByAge(sessionStateDir, 30); // 30 zile
const rec = pruneDirByAge(recoveryDir, 7);     // 7 zile
log(`prune: session-state removed=${sst.removed}, kept=${sst.kept}`);
log(`prune: recovery removed=${rec.removed}, kept=${rec.kept}`);
```
